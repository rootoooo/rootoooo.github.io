    <script>
        // 获取当前用户的 IP 和国家信息
        fetch('https://api.ipify.org?format=json')
            .then(response => response.json())
            .then(data => {
                document.getElementById('ip').textContent = data.ip;
                fetch(`https://ipinfo.io/${data.ip}/json`)
                    .then(response => response.json())
                    .then(info => {
                        document.getElementById('country').textContent = info.country;
                    });
            })
            .catch(error => {
                document.getElementById('ip').textContent = '无法获取 IP';
                document.getElementById('country').textContent = '无法获取国家';
            });
